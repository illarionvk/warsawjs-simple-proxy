{% highlight javascript linenos %}
server.route({
  method: 'GET',
  path: '/instagram.json',
  handler: function(request, reply) {
    return server.methods.getInstagram(function(error, result) {
      if (error) {
        return reply({
          error: error
        }).code(500);
      }
      return reply(result);
    });
  },
  config: {
    cors: true,
    cache: {
      expiresIn: 60 * 60 * 1000, // One hour
      privacy: 'private'
    }
  }
});
{% endhighlight %}
